def doDELETE(path,header=nil) 
    server='https://'+resource[:transport]['server']+':'+resource[:transport]['port'].to_s+path
    uri = URI.parse(server)
    http = Net::HTTP.new(uri.host, uri.port)
    http.use_ssl = true
    http.verify_mode = OpenSSL::SSL::VERIFY_NONE
    request = Net::HTTP::Delete.new(uri.request_uri,header)
    response = http.request(request)
    case response
        when Net::HTTPSuccess
  	        return true
        when Net::HTTPUnauthorized
            raise "Unauthorized"
	    else
  	        raise response.message
    end
    return false    
end

def returnAllPerfPolicies
  $token=getAuthToken
  allPolicies = Array.new
  totalPolicies    = doGET("/v1/performance_policies",{"X-Auth-Token"=>$token})

  startRow   = 0
  endRow     = totalPolicies["endRow"]
  totalRows  = totalPolicies["totalRows"]

  # Do pagination
  begin
    policies = doGET("/v1/performance_policies?startRow="+startRow.to_s,{"X-Auth-Token"=>$token})["data"]
    policies.each do |policy|
      allPolicies.push(policy)
    end
    startRow = startRow + policies.size
  end while (allPolicies.size < totalRows)
  return allPolicies
end

def returnPerfPolicyId(perfPolicyName)
  $token=getAuthToken
  allPerfPolicies = returnAllPerfPolicies
  allPerfPolicies.each do |policy|
      if perfPolicyName.eql? policy["name"]
        return policy["id"]
      end
  end
  return nil
end
